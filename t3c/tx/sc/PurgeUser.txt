# Manage-UserProfiles.ps1 — Interaktives Aufräumen lokaler Profile (PS 5.1)
[CmdletBinding(SupportsShouldProcess = $true)]
param(
    [switch]$Quiet
)

function Assert-Admin {
    $isAdmin = ([Security.Principal.WindowsPrincipal] [Security.Principal.WindowsIdentity]::GetCurrent()
               ).IsInRole([Security.Principal.WindowsBuiltInRole]::Administrator)
    if (-not $isAdmin) { Write-Warning "Bitte als Administrator ausführen. Skript wird beendet."; exit 1 }
}

function Try-TranslateSid([string]$Sid){
    try { (New-Object System.Security.Principal.SecurityIdentifier($Sid)).Translate([System.Security.Principal.NTAccount]).Value } catch { $null }
}

function Convert-LastUse([string]$Dmtf){
    if ([string]::IsNullOrWhiteSpace($Dmtf)) { return $null }
    if ($Dmtf -notmatch '^\d{14}\.\d{6}[+\-]\d{3}$') { return $null }
    try { [Management.ManagementDateTimeConverter]::ToDateTime($Dmtf) } catch { $null }
}

function Normalize-Input([string]$s){
    if ($null -eq $s) { return '' }
    $s = ($s.ToCharArray() | Where-Object { [int]$_ -gt 31 -or [int]$_ -in 9,10,13 }) -join ''
    $s = $s -replace '[;\u00B7\u0387\uFF1B\uFF0C\u060C\u061B\u1363\u1802\u1808\uFE10\uFE11\uFE14\uFE50\uFE51\uFE54]', ','
    $s = $s -replace '[-\u2010\u2011\u2012\u2013\u2014\u2212\uFE58\uFF0D]', '-'
    $s = $s -replace '\s*,\s*', ','
    $s = $s -replace '\s*-\s*', '-'
    $s = $s.Trim()
    return $s
}

function Parse-Selection([string]$text){
    $norm = Normalize-Input $text
    $nums = New-Object System.Collections.Generic.List[int]
    $rangeRe = [regex]'(?<!\d)(\d+)-(\d+)(?!\d)'
    $rangeMatches = $rangeRe.Matches($norm)
    foreach($m in $rangeMatches){
        $a = [int]$m.Groups[1].Value; $b = [int]$m.Groups[2].Value
        if ($a -le $b){ ($a..$b) | ForEach-Object { [void]$nums.Add($_) } }
        else          { ($b..$a) | ForEach-Object { [void]$nums.Add($_) } }
    }
    $singleRe = [regex]'\d+'
    foreach($m in $singleRe.Matches($norm)){ [void]$nums.Add([int]$m.Value) }
    $out = $nums.ToArray() | Select-Object -Unique | Sort-Object

    Write-Host "DEBUG Raw Eingabe     : $text" -ForegroundColor DarkGray
    Write-Host "DEBUG Normalisiert    : $norm" -ForegroundColor DarkGray
    if ($rangeMatches.Count -gt 0) {
        $rangesStr = ($rangeMatches | ForEach-Object { "$($_.Groups[1].Value)-$($_.Groups[2].Value)" }) -join ', '
        Write-Host "DEBUG Erkannte Ranges : $rangesStr" -ForegroundColor DarkGray
    } else { Write-Host "DEBUG Erkannte Ranges : (keine)" -ForegroundColor DarkGray }
    $allSingles = ([regex]::Matches($norm,'\d+') | ForEach-Object { $_.Value }) -join ', '
    Write-Host "DEBUG Erkannte Zahlen : " -NoNewline -ForegroundColor DarkGray
    if ($allSingles) { Write-Host $allSingles -ForegroundColor DarkGray } else { Write-Host "(keine)" -ForegroundColor DarkGray }

    return ,$out
}

function Remove-UserProfileRobust {
    param(
        [Parameter(Mandatory)] [string]$Sid,
        [Parameter(Mandatory)] $CimInstance,
        [Parameter(Mandatory)] [string]$Path,
        [switch]$Quiet
    )
    try {
        $ret = Invoke-CimMethod -InputObject $CimInstance -MethodName Delete -ErrorAction Stop
        $code = if ($ret -and $ret.ReturnValue -ne $null) { [int]$ret.ReturnValue } else { 0 }
        if ($code -eq 0) { return @{OK=$true; Code=$code; Err=$null} }
    } catch {}

    try {
        Remove-CimInstance -InputObject $CimInstance -ErrorAction Stop
        return @{OK=$true; Code=0; Err=$null}
    } catch {}

    try {
        $wmi = Get-WmiObject -Class Win32_UserProfile -Namespace root\cimv2 -Filter ("SID='{0}'" -f $Sid) -ErrorAction Stop
        if ($wmi) {
            $null = $wmi.Delete()
            return @{OK=$true; Code=0; Err=$null}
        } else {
            return @{OK=$false; Code=$null; Err="WMI-Objekt nicht gefunden"}
        }
    } catch {
        return @{OK=$false; Code=$null; Err=$_.Exception.Message}
    }
}

Assert-Admin

$currentSid = [System.Security.Principal.WindowsIdentity]::GetCurrent().User.Value
$currentUser= [System.Security.Principal.WindowsIdentity]::GetCurrent().Name

$profilesRaw = Get-CimInstance -ClassName Win32_UserProfile | Where-Object {
    $_.LocalPath -and $_.LocalPath -like "$($env:SystemDrive)\Users\*" -and -not $_.Special
}

if (-not $profilesRaw) { Write-Host "Keine benutzerbezogenen Profile gefunden." -ForegroundColor Yellow; return }

$idx = 0
$profiles = $profilesRaw | Sort-Object LastUseTime | ForEach-Object {
    $idx++
    $sid  = $_.SID
    $acct = Try-TranslateSid $sid
    $name = if ($acct) { $acct } else { Split-Path -Path $_.LocalPath -Leaf }
    New-Object psobject -Property @{
        Nr            = $idx
        Name          = $name
        SID           = $sid
        Pfad          = $_.LocalPath
        Geladen       = [bool]$_.Loaded
        LetzteNutzung = Convert-LastUse $_.LastUseTime
        CimRef        = $_
    }
}

if (-not $Quiet){
    Write-Host "`nGefundene Profile:" -ForegroundColor Cyan
    $profiles | Select-Object Nr, Name, Pfad, Geladen, LetzteNutzung | Format-Table -AutoSize
}

Write-Host "`nGib die NUMMERN der Profile ein, die ERHALTEN bleiben sollen (z. B. 1,2,4 oder 1-3,5)." -ForegroundColor Cyan
$selection = Read-Host "Nummern behalten"
if ([string]::IsNullOrWhiteSpace($selection)) { Write-Host "Abgebrochen." -ForegroundColor Yellow; return }

[int[]]$keepNums = Parse-Selection -text $selection
$keepStr = if ($keepNums -and $keepNums.Count -gt 0) { ($keepNums -join ',') } else { '(leer)' }
Write-Host "Eingabe interpretiert als BEHALTEN: $keepStr" -ForegroundColor DarkCyan

$keepMap = @{}
foreach($n in $keepNums){ $keepMap[[int]$n] = $true }

$kept     = $profiles | Where-Object { $keepMap.ContainsKey([int]$_.Nr) -or ($_.SID -eq $currentSid) }
$toDelete = $profiles | Where-Object { -not $keepMap.ContainsKey([int]$_.Nr) -and ($_.SID -ne $currentSid) }

Write-Host "`nDiese Profile werden BEHALTEN:" -ForegroundColor Green
$kept | Select-Object Nr, Name, Pfad, Geladen | Format-Table -AutoSize

if (-not $toDelete){ Write-Host "Es bleibt alles erhalten – nichts zu löschen." -ForegroundColor Green; return }

Write-Host "`nFolgende Profile würden GELÖSCHT:" -ForegroundColor Red
$toDelete | Select-Object Nr, Name, Pfad, Geladen, LetzteNutzung | Format-Table -AutoSize

$loaded = $toDelete | Where-Object { $_.Geladen }
if ($loaded){
    Write-Warning "Einige Profile sind geladen und können jetzt nicht gelöscht werden:"
    $loaded | Select-Object Nr, Name, Pfad | Format-Table -AutoSize
    $toDelete = $toDelete | Where-Object { -not $_.Geladen }
    if (-not $toDelete){ Write-Host "Es bleibt nichts übrig, was gelöscht werden kann." -ForegroundColor Yellow; return }
}

$cnt = $toDelete.Count
$confirm = Read-Host "`n$cnt Profile löschen? (J/N)"
if ($confirm -notin @('J','j','Y','y')) { Write-Host "Abgebrochen." -ForegroundColor Yellow; return }

$results = @()
foreach ($p in $toDelete) {
    $action = "Profil löschen: $($p.Name) [$($p.SID)] - $($p.Pfad)"
    if ($PSCmdlet.ShouldProcess($p.Pfad, "Delete user profile")) {
        $res = Remove-UserProfileRobust -Sid $p.SID -CimInstance $p.CimRef -Path $p.Pfad -Quiet:$Quiet
        $results += New-Object psobject -Property @{
            Name   = $p.Name; Pfad=$p.Pfad; SID=$p.SID
            Erfolg = $res.OK; Code=$res.Code; Fehler=$res.Err
        }
        if ($res.OK) { Write-Host "[OK] $action" -ForegroundColor Green }
        else         { Write-Warning "[FEHLER] $action : $($res.Err)" }
    }
}

$okCount = ($results | Where-Object { $_.Erfolg }).Count
$errCount = $results.Count - $okCount
Write-Host "`nFertig. Erfolgreich gelöscht: $okCount, Fehler: $errCount" -ForegroundColor Cyan
if (-not $Quiet) { $results | Format-Table Name, Pfad, SID, Erfolg, Code, Fehler -AutoSize }

# ===== TXT-Log schreiben =====
$logDir = "C:\Temp"
$logFile = Join-Path $logDir "Profdel.txt"
if (-not (Test-Path $logDir)) { New-Item -Path $logDir -ItemType Directory | Out-Null }

$ts   = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
$hostnm = $env:COMPUTERNAME
$succ = $results | Where-Object { $_.Erfolg }

$lines = @()
$lines += $ts
if ($succ.Count -gt 0) {
    $lines += "Auf $hostnm wurden die folgenden Profile erfolgreich durch $currentUser entfernt:"
    $lines += ($succ | ForEach-Object { $_.Name })
} else {
    $lines += "Auf $hostnm wurden keine Profile entfernt (Ausgeführt durch $currentUser)."
}
$lines | Out-File -FilePath $logFile -Encoding UTF8

Write-Host "Log geschrieben nach $logFile" -ForegroundColor Cyan
